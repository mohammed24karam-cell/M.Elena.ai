def apply_style(gender):
    if gender == "بنت":
        bg, text = "#FFF0F5", "#D81B60"
    elif gender == "شاب":
        bg, text = "#E1F5FE", "#01579B"
    else:
        bg, text = "#FFFFFF", "#333333"
    
    st.markdown(f"""<style>.stApp {{ background-color: {bg}; color: {text}; }}</style>""", unsafe_allow_html=True)

# التعليمات البرمجية للشخصية (السر خلف اللباقة)
system_prompt = f"""
أنت مساعد ذكي جداً تعرف كل شيء في العالم. 
أسلوبك: هادئ، لطيف، محترم جداً ولبق.
إذا كان المستخدم 'بنت': كن أنثوياً في تعبيراتك (مثلاً: أهلاً بكِ عزيزتي، يسعدني مساعدتكِ).
إذا كان المستخدم 'شاب': كن هادئاً وواقعياً.
تحدث حصراً بلغة: {lang}.
أنت بارع في حل الرياضيات وكشف فبركة الصور.
"""

if "messages" not in st.session_state:
    st.session_state.messages = []

# عرض الرسائل القديمة
for m in st.session_state.messages:
    with st.chat_message(m["role"]):
        st.markdown(m["content"])

# إدخال الصور أو الأسئلة
img_file = st.file_uploader("ارفع صورة للتحليل (رياضيات/فبركة)", type=['png', 'jpg', 'jpeg'])
user_input = st.chat_input("اسألني أي شيء...")

if user_input:
    st.session_state.messages.append({"role": "user", "content": user_input})
    with st.chat_message("user"):
        st.write(user_input)

    # إرسال الطلب لـ Gemini
    with st.chat_message("assistant"):
        prompt_parts = [system_prompt, user_input]
        if img_file:
            img = Image.open(img_file)
            prompt_parts.append(img)
        
        response = model.generate_content(prompt_parts)
        st.write(response.text)
        st.session_state.messages.append({"role": "assistant", "content": response.text})
